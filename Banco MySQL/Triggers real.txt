CREATE DATABASE Aprendendo_TRIGGER;

CREATE TABLE produto(
id_produto INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
nome VARCHAR(256),
preco DECIMAL(8.2),
estoque INT
);

CREATE TABLE pedido(
id_pedido INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
data_horario DATETIME,
quantidade INT,
produto_id INT,
FOREIGN KEY (produto_id) REFERENCES produto (id_produto)
);

DELIMITER $$
CREATE TRIGGER estoque_atualizado
AFTER INSERT ON pedido -- depois de inserir na tabela pedido
FOR EACH ROW -- para cada linha
BEGIN -- comece
UPDATE produto -- atualize
SET produto.estoque = produto.estoque - new.quantidade -- setar no estoque o estoque menos a quantidade pedida pelo cliente
WHERE produto.id_produto = new.produto_id; 
END;
$$
